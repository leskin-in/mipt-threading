data_1 = []
data_2 = []

counter = 0
with open("stats.txt", "r") as f:
    for line in f:
        counter += 1
        if (counter == DATA_1_SIZE + 1):
            continue
        temp_list = line.split(" ")
        temp_list[2] = temp_list[2][:-1]
        if (counter < 6):
            data_1.append(temp_list)
        else:
            data_2.append(temp_list)

data_1_N = [int(data_1[i][6]) for i in range(DATA_1_SIZE)]
data_1_T = [float(data_1[i][2]) for i in range(DATA_1_SIZE)]
data_1_S = [float(data_1[i][1]) for i in range(DATA_1_SIZE)]
data_1_E = [float(data_1[i][0]) for i in range(DATA_1_SIZE)]

data_2_P = [int(data_2[i][8]) for i in range(DATA_2_SIZE)]
data_2_T = [float(data_2[i][2]) for i in range(DATA_2_SIZE)]
data_2_S = [float(data_2[i][1]) for i in range(DATA_2_SIZE)]
data_2_E = [float(data_2[i][0]) for i in range(DATA_2_SIZE)]

Построим графики для первого набора данных (зависимости от $N$):
